Goal_Value_rank <- score_manager %>% 
  filter(CP == 0) %>% 
  gather(key = "score", value = "scorer", #scorerをまとめる
         starts_with("score")) %>% 
  filter(scorer != "失点") %>%

  mutate(No. = row_number()) %>% 
  left_join(Value_of_theGoal, by = "No.") %>% 
  group_by(scorer) %>% 
  summarise(n = n(),

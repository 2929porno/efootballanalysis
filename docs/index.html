Value_of_theGoal <- score %>% 
  filter(CP == 0) %>%
  group_by(GF) %>% 
  summarise(W = mean(W)) %>% 
  mutate(lag = lag(W),
         marginal = W-lag(W)) %>% 
  replace_na(replace = list(marginal = 0)) %>% 
  rename("No." = GF)

#GFのWに対する相対度数の微分（限界効果を計算）
Value_of_theGoal %>% 
  ggplot(aes(No.,marginal))+
  geom_point()+
  geom_line()+
  scale_x_continuous(breaks = seq(0,8,by=1))
